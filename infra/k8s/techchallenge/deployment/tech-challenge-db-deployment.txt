apiVersion: apps/v1
kind: Deployment
metadata:
  name: tech-challenge-db
  namespace: techchallenge
  labels:
    app: tech-challenge-db
spec:
  replicas: 1
  selector:
    matchLabels:
      app: tech-challenge-db
  template:
    metadata:
      labels:
        app: tech-challenge-db
    spec:
      containers:
        - name: tech-challenge-db
          image: postgres:13
          ports:
            - containerPort: 5432
          env:
            - name: POSTGRES_DB
              valueFrom:
                secretKeyRef:
                  name: tech-challenge-secret
                  key: db-name
            - name: POSTGRES_USER
              valueFrom:
                secretKeyRef:
                  name: tech-challenge-secret
                  key: db-username
            - name: POSTGRES_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: tech-challenge-secret
                  key: db-password
